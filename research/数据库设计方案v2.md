# 家庭记账系统 - 数据库设计方案 v2

> 更新时间:2026-07-15
> 更新内容:新增「账本」模块,支持可分可合的灵活分析

---

## 设计原则

1. **所有账单在一个大池子里** - 数据不隔离,通过标签维度实现分组
2. **账本 = 虚拟分组标签** - 不是物理隔离,是逻辑聚合
3. **随时可分可合** - 单账本分析、多账本合并、全局汇总,任意切换
4. **三维账单模型** - 资金来源 × 支付渠道 × 交易平台
5. **快速录入 + 延迟完善** - 先记最简信息,后续导入补充
6. **双式记账法** - 用户无感,系统自动维护借贷平衡
7. **多币种支持** - 出国旅行场景,保留原始金额 + 自动换算人民币
8. **离线优先** - 手机端本地SQLite存储,有网时自动同步,无网也能用

---

```sql
CREATE TABLE account_books (

-- 每个家庭只有一个默认账本的约束(用部分唯一索引)
```

### 3. 交易账单(核心表)

```sql
-- ============================================================
-- 交易记录(核心表) - 精简版,仅保留高频使用字段
-- 采用双式记账法,每笔交易自动生成借贷两条记录
-- 外币结算和退款追踪拆分到独立扩展表
-- 按 transaction_time 季度分区,查询自动裁剪无关分区
-- ============================================================
CREATE TABLE transactions (
    id                  BIGSERIAL,
    family_id           BIGINT NOT NULL REFERENCES families(id),
    book_id             BIGINT NOT NULL REFERENCES account_books(id),

    -- 双式记账关联
    entry_id            BIGINT,                     -- 同一笔交易的借方贷方共享
    entry_side          VARCHAR(4) NOT NULL,        -- debit / credit

    -- 基础信息(必填)
    type                VARCHAR(20) NOT NULL,       -- expense / income / transfer
    amount              BIGINT NOT NULL,             -- 金额(单位:分,35元=3500,避免浮点精度问题)
    currency            VARCHAR(3) DEFAULT 'CNY',   -- 币种代码(ISO 4217)

    -- 外币原始金额(仅外币交易时填写,人民币交易为NULL)
    original_amount     BIGINT,                     -- 原始外币金额(单位:分)
    original_currency   VARCHAR(3),                 -- 原始币种
    exchange_rate       DECIMAL(10,6),              -- 记账时汇率(非金额,用DECIMAL)

    -- 分类体系(三级)
    category_id         BIGINT REFERENCES categories(id),
    sub_category_id     BIGINT REFERENCES categories(id),
    detail_category_id  BIGINT REFERENCES categories(id),

    -- 三维账单模型
    payment_account_id  BIGINT REFERENCES payment_accounts(id),
    payment_channel_id  BIGINT REFERENCES payment_channels(id),
    platform_id         BIGINT REFERENCES platforms(id),

    -- 商户信息
    merchant_name       VARCHAR(200),
    description         VARCHAR(500),

    -- 时间
    transaction_time    TIMESTAMPTZ NOT NULL,
    recorded_at         TIMESTAMPTZ DEFAULT NOW(),

    -- 记录人
    recorded_by         BIGINT NOT NULL REFERENCES users(id),
    paid_by             BIGINT REFERENCES users(id),

    -- 快速录入标记
    is_quick_entry      BOOLEAN DEFAULT FALSE,
    completion_status   VARCHAR(20) DEFAULT 'complete',

    -- 周期性交易
    recurring_id        BIGINT REFERENCES recurring_transactions(id),

    -- 原始数据
    raw_data            JSONB,
    import_id           BIGINT REFERENCES bill_imports(id),

    -- 全文搜索（pg_trgm GIN索引，或zhparser tsvector，详见策略三）
    -- search_vector  tsvector,  -- 仅zhparser方案需要

    -- 同步字段(详见16.离线同步机制)
    version             INTEGER NOT NULL DEFAULT 1,
    is_deleted          BOOLEAN DEFAULT FALSE,
    created_at          TIMESTAMPTZ DEFAULT NOW(),
    updated_at          TIMESTAMPTZ DEFAULT NOW(),

    -- 主键(分区表必须包含分区键)
    PRIMARY KEY (id, transaction_time),

    -- CHECK 约束
    CONSTRAINT chk_entry_side CHECK (entry_side IN ('debit', 'credit')),
    CONSTRAINT chk_type CHECK (type IN ('expense', 'income', 'transfer')),
    CONSTRAINT chk_amount CHECK (amount > 0),
    CONSTRAINT chk_completion CHECK (completion_status IN ('complete', 'pending', 'partial'))
) PARTITION BY RANGE (transaction_time);

-- 按季度创建分区(后续由定时任务自动创建)
CREATE TABLE transactions_2026q1 PARTITION OF transactions
    FOR VALUES FROM ('2026-01-01') TO ('2026-04-01');
CREATE TABLE transactions_2026q2 PARTITION OF transactions
    FOR VALUES FROM ('2026-04-01') TO ('2026-07-01');
CREATE TABLE transactions_2026q3 PARTITION OF transactions
    FOR VALUES FROM ('2026-07-01') TO ('2026-10-01');
CREATE TABLE transactions_2026q4 PARTITION OF transactions
    FOR VALUES FROM ('2026-10-01') TO ('2027-01-01');

-- 索引(在分区表上创建,自动应用到所有分区)
CREATE INDEX idx_txn_family ON transactions(family_id);
CREATE INDEX idx_txn_book ON transactions(book_id);
CREATE INDEX idx_txn_time ON transactions(transaction_time);
CREATE INDEX idx_txn_category ON transactions(category_id);
CREATE INDEX idx_txn_status ON transactions(completion_status);
CREATE INDEX idx_txn_entry ON transactions(entry_id);
CREATE INDEX idx_txn_account ON transactions(payment_account_id);
CREATE INDEX idx_txn_recorded_by ON transactions(recorded_by);
CREATE INDEX idx_txn_paid_by ON transactions(paid_by);
CREATE INDEX idx_txn_import ON transactions(import_id);
CREATE INDEX idx_txn_recurring ON transactions(recurring_id);
```

**双式记账 - 转账类型的处理:**
```
场景:从支付宝余额转1000元到招商储蓄卡

普通消费(expense/income):产生2条记录
  借方:钱去哪了(分类)
  贷方:钱从哪来(账户)

转账(transfer):产生2条记录(不是4条)
  借方:目标账户(招商储蓄卡) +1000  entry_side='debit'
  贷方:来源账户(支付宝余额) -1000  entry_side='credit'
  type 都是 'transfer'
  category_id 为 NULL(转账没有分类)

┌─────────────────────────────────────────────────────┐
│ 记录A(借方)                                          │
│ id: 2001, entry_id: 1000, entry_side: debit          │
│ type: transfer, amount: 1000.00                      │
│ payment_account_id: 招商储蓄卡(钱到哪了)            │
│ category_id: NULL(转账无分类)                        │
├─────────────────────────────────────────────────────┤
│ 记录B(贷方)                                          │
│ id: 2002, entry_id: 1000, entry_side: credit         │
│ type: transfer, amount: 1000.00                      │
│ payment_account_id: 支付宝余额(钱从哪来)            │
│ category_id: NULL(转账无分类)                        │
└─────────────────────────────────────────────────────┘

效果:
  支付宝余额: -1000(贷方扣款)
  招商储蓄卡: +1000(借方入账)
  总资产不变
```

**账户余额自动计算:**
```sql
-- 每个账户的精确余额(无需手动维护)
SELECT
    pa.id,
    pa.name,
    pa.initial_balance
    + COALESCE(SUM(CASE WHEN t.entry_side='debit' AND t.type='income' THEN t.amount ELSE 0 END), 0)   -- 收入进账
    - COALESCE(SUM(CASE WHEN t.entry_side='credit' AND t.type='expense' THEN t.amount ELSE 0 END), 0) -- 支出扣款
    + COALESCE(SUM(CASE WHEN t.entry_side='debit' AND t.type='transfer' THEN t.amount ELSE 0 END), 0) -- 转入
    - COALESCE(SUM(CASE WHEN t.entry_side='credit' AND t.type='transfer' THEN t.amount ELSE 0 END), 0) -- 转出
    AS calculated_balance
FROM payment_accounts pa
LEFT JOIN transactions t ON t.payment_account_id = pa.id AND t.is_deleted = FALSE
GROUP BY pa.id, pa.name, pa.initial_balance;
```


```
    │
    ▼
┌─ 第一步:记账时(用户或自动记账) ─────────────────────────┐
│ original_amount:   500                                      │
│ original_currency: JPY                                      │
│ exchange_rate:     0.048 (当天市场汇率,自动获取或手动输入)│
│ amount:            500 × 0.048 = 24.00 CNY                  │
│ settlement_status: pending (还未结算)                      │
│ settlement_amount: NULL                                      │
│ settlement_rate:   NULL                                      │
│ settlement_date:   NULL                                      │
│                                                              │
│ → 用途:实时记账、旅行预算追踪、当月报表                    │
│ → 按 24.00 CNY 统计                                         │
└──────────────────────────────────────────────────────────────┘
    │
    │  1-3天后,银行入账
    ▼
┌─ 第二步:银行入账后(导入信用卡账单时更新) ────────────────┐
│ settlement_amount: 24.10 (银行账单显示的实际人民币金额)    │
│ settlement_rate:   0.0482 (银行入账汇率)                   │
│ settlement_date:   2026-07-12                               │
│ settlement_status: settled                                   │
│                                                              │
│ → amount 保持 24.00 不变(记账时的快照)                     │
│ → 结算数据存入 settlement_* 字段                             │
│ → 差额 0.10 元 = 汇率波动差异                                │
└──────────────────────────────────────────────────────────────┘
    │
    │  8月5日,信用卡还款
    ▼
┌─ 第三步:还款时 ────────────────────────────────────────────┐
│ 还款操作产生一条 transfer 记录:                             │
│   从:银行储蓄卡 → 到:信用卡                               │
│   金额:24.10 CNY(用结算金额还款)                          │
│                                                              │
│ → 如果还款时汇率又有变化,那是还款的汇率,不影响消费记录    │
│ → 信用卡账单已锁定金额,还款只是转账操作                    │
└──────────────────────────────────────────────────────────────┘
```

**三种统计口径(用户可切换):**

```sql
-- 口径1:按记账时金额(实时,适合旅行中看花了多少)
SELECT SUM(amount) as total FROM transactions WHERE type='expense';
-- 结果:24.00 CNY

-- 口径2:按结算金额(精确,适合最终对账)
SELECT SUM(COALESCE(settlement_amount, amount)) as total FROM transactions WHERE type='expense';
-- 结果:24.10 CNY(用了银行实际入账金额)

-- 口径3:按原始币种(看外币总花费)
SELECT original_currency, SUM(original_amount) as total
FROM transactions WHERE original_currency IS NOT NULL GROUP BY original_currency;
-- 结果:JPY 500
```

```sql
    original_amount,
    original_currency,
    amount as 记账金额_cny,
```

**自动汇率获取:**

```python
# 每天凌晨自动拉取汇率并存入 exchange_rates 表
async def fetch_and_store_rates():
    # 从 API 获取最新汇率
    rates = await fetch_from_api("exchangerate-api.com")

    for currency in ["USD", "EUR", "JPY", "KRW", "HKD", "GBP", "THB"]:
        await db.execute("""
            INSERT INTO exchange_rates (base_currency, target_currency, rate, source, rate_date)
            VALUES ($1, 'CNY', $2, 'api', CURRENT_DATE)
            ON CONFLICT (base_currency, target_currency, rate_type, rate_date)
            DO UPDATE SET rate = $2
        """, currency, rates[currency])

# 记账时自动获取当天汇率
async def get_rate(from_currency: str, to_currency: str = 'CNY') -> float:
    rate = await db.fetchval("""
        SELECT rate FROM exchange_rates
        WHERE base_currency = $1 AND target_currency = $2 AND rate_date = CURRENT_DATE
        ORDER BY created_at DESC LIMIT 1
    """, from_currency, to_currency)

    if not rate:
        # 如果今天没有,取最近一天的
        rate = await db.fetchval("""
            SELECT rate FROM exchange_rates
            WHERE base_currency = $1 AND target_currency = $2
            ORDER BY rate_date DESC LIMIT 1
        """, from_currency, to_currency)

    return rate
```

**特殊场景处理:**

```
场景1:信用卡外币消费
  7/10消费500 JPY → 记账时按当天汇率24.00 CNY
  7/12银行入账 → 导入账单时更新 settlement_amount=24.10 CNY
  8/5还款 → 还款24.10 CNY(用结算金额)

场景2:现金外币消费
  在日本用现金买拉面 1200 JPY
  → 兑换时已确定汇率(买入日元时的汇率)
  → exchange_rate 用兑换时的汇率
  → settlement_status 直接标记为 settled(现金即时结算)

场景3:支付宝/微信境外消费
  支付宝在泰国消费 500 THB
  → 支付宝按实时汇率扣款,直接扣人民币
  → amount 直接用支付宝扣款金额
  → 无需外币字段

场景4:同一笔交易多次汇率更新
  7/10记账:汇率0.048 → amount=24.00
  7/12入账:汇率0.0482 → settlement_amount=24.10
  7/15发现入账有误,银行更正:汇率0.0481 → settlement_amount=24.05
  → 每次更新只改 settlement_* 字段,保留历史记录
```

### 16. 离线同步机制(重设计)

**设计目标:** 解决旧方案的三个核心问题——时钟偏差导致丢数据、双式记账原子性、并发编辑冲突检测。

**核心思路:** 基于**变更日志(Change Log)** + **逻辑序列号** + **记录版本号**实现可靠同步。

#### 16.1 服务器端同步表

```sql
-- ============================================================
-- 家庭级全局序列号:每产生一条变更日志自增,用于增量拉取
-- ============================================================
CREATE TABLE family_sync_seq (
    family_id       BIGINT PRIMARY KEY REFERENCES families(id),
    current_seq     BIGINT NOT NULL DEFAULT 0
);

-- ============================================================
-- 变更日志:所有可同步表的写操作都记录在此
-- 客户端通过 last_seq 增量拉取,不依赖时间戳
-- ============================================================
CREATE TABLE sync_change_log (
    id              BIGSERIAL PRIMARY KEY,
    family_id       BIGINT NOT NULL REFERENCES families(id),
    seq             BIGINT NOT NULL,                 -- 家庭级递增序列号(拉取游标)
    table_name      VARCHAR(50) NOT NULL,            -- 变更的表名
    record_id       BIGINT NOT NULL,                 -- 变更记录的ID
    operation       VARCHAR(10) NOT NULL,             -- INSERT / UPDATE / DELETE
    version         INTEGER NOT NULL,                -- 记录版本号(用于冲突检测)
    changed_by      BIGINT REFERENCES users(id),     -- 可为NULL(系统预设数据变更时)
    changed_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    change_data     JSONB,                           -- 变更后的完整记录快照(INSERT/UPDATE时)
    family_id_check BIGINT NOT NULL,                 -- 冗余,便于分区和校验

    CONSTRAINT chk_sync_op CHECK (operation IN ('INSERT', 'UPDATE', 'DELETE'))
);

CREATE INDEX idx_sync_log_family_seq ON sync_change_log(family_id, seq);
CREATE INDEX idx_sync_log_table ON sync_change_log(family_id, table_name, seq);

-- 存储优化:定期清理过期变更日志(客户端90天内必须同步一次)
-- 建议按月分区,PARTITION BY RANGE(changed_at),过期分区直接DROP(避免VACUUM开销)
-- 可选:change_data 只存变更字段(增量diff)而非完整快照,进一步节省空间

-- ============================================================
-- 客户端同步状态:记录每个设备同步到了哪条序列号
-- ============================================================
CREATE TABLE client_sync_state (
    id              BIGSERIAL PRIMARY KEY,
    client_id       VARCHAR(100) NOT NULL,           -- 客户端设备唯一ID(UUID)
    family_id       BIGINT NOT NULL REFERENCES families(id),
    user_id         BIGINT NOT NULL REFERENCES users(id),
    last_pushed_seq BIGINT DEFAULT 0,                -- 客户端已推送到的序列号
    last_pulled_seq BIGINT DEFAULT 0,                -- 客户端已拉取到的序列号
    device_type     VARCHAR(20),                     -- android / ios / web
    device_name     VARCHAR(100),
    app_version     VARCHAR(20),
    last_active_at  TIMESTAMPTZ DEFAULT NOW(),
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(client_id, family_id)
);

-- ============================================================
-- 同步日志:每次同步操作的审计记录
-- ============================================================
CREATE TABLE sync_logs (
    id              BIGSERIAL PRIMARY KEY,
    client_id       VARCHAR(100) NOT NULL,
    family_id       BIGINT NOT NULL REFERENCES families(id),
    sync_type       VARCHAR(20) NOT NULL,            -- push / pull
    record_count    INTEGER DEFAULT 0,
    conflict_count  INTEGER DEFAULT 0,
    status          VARCHAR(20) DEFAULT 'success',   -- success / partial / failed
    error_message   VARCHAR(500),
    duration_ms     INTEGER,
    created_at      TIMESTAMPTZ DEFAULT NOW()
```

#### 16.3 变更日志自动生成(触发器)

```sql
-- 通用触发器函数:任何可同步表的INSERT/UPDATE/DELETE都自动写入sync_change_log
CREATE OR REPLACE FUNCTION fn_record_sync_change()
RETURNS TRIGGER AS $$
DECLARE
    v_family_id BIGINT;
    v_seq       BIGINT;
    v_operation VARCHAR(10);
    v_version   INTEGER;
    v_data      JSONB;
    v_record_id BIGINT;
BEGIN
    -- 确定操作类型和数据
    IF TG_OP = 'DELETE' THEN
        v_operation := 'DELETE';
        v_record_id := OLD.id;
        v_family_id := OLD.family_id;
        v_version   := 1;  -- 删除不需要版本
        v_data      := NULL;
    ELSIF TG_OP = 'UPDATE' THEN
        -- 如果只是更新updated_at或version,不记录(避免循环触发)
        IF OLD.version = NEW.version THEN
            RETURN NEW;
        END IF;
        v_operation := 'UPDATE';
        v_record_id := NEW.id;
        v_family_id := NEW.family_id;
        v_version   := NEW.version;
        v_data      := to_jsonb(NEW);
    ELSE
        v_operation := 'INSERT';
        v_record_id := NEW.id;
        v_family_id := NEW.family_id;
        v_version   := NEW.version;
        v_data      := to_jsonb(NEW);
    END IF;

    -- 获取下一个序列号
    UPDATE family_sync_seq
    SET current_seq = current_seq + 1
    WHERE family_id = v_family_id
    RETURNING current_seq INTO v_seq;

    IF v_seq IS NULL THEN
        -- 首次同步,初始化序列
        INSERT INTO family_sync_seq (family_id, current_seq)
        VALUES (v_family_id, 1)
        ON CONFLICT (family_id) DO UPDATE SET current_seq = family_sync_seq.current_seq + 1
        RETURNING current_seq INTO v_seq;
    END IF;

    -- 写入变更日志
    INSERT INTO sync_change_log
        (family_id, seq, table_name, record_id, operation, version, changed_by, change_data, family_id_check)
    VALUES
        (v_family_id, v_seq, TG_TABLE_NAME, v_record_id, v_operation, v_version,
         CASE WHEN COALESCE(NEW.recorded_by, NEW.created_by, NEW.user_id, 0) = 0
              THEN NULL
              ELSE COALESCE(NEW.recorded_by, NEW.created_by, NEW.user_id)
         END,
         v_data, v_family_id);

    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;
```

```
手机端                                    服务器端
  │                                         │
  │  ┌──── 阶段1: PUSH(上传本地变更) ────┐  │
  │  │                                    │  │
  │  │  POST /api/sync/push               │  │
  │  │  {                                 │  │
  │  │    client_id: "device-uuid",       │  │
  │  │    base_seq: 42,  ← 上次拉取到的  │  │
  │  │    changes: [                      │  │
  │  │      {                             │  │
  │  │        table: "transactions",      │  │
  │  │        local_id: "local-uuid-1",   │  │
  │  │        operation: "INSERT",        │  │
  │  │        base_version: 0,            │  │
  │  │        data: { amount, type, ... } │  │
  │  │      },                           │  │
  │  │      {                             │  │
  │  │        table: "transactions",      │  │
  │  │        local_id: "local-uuid-2",   │  │
  │  │        operation: "UPDATE",        │  │
  │  │        base_version: 3, ← 修改前  │  │
  │  │        data: { amount: 50, ... }   │  │
  │  │      }                             │  │
  │  │    ]                               │  │
  │  │  }                                 │  │
  │  │                                    │  │
  │  │  服务器处理:                       │  │
  │  │  ├── INSERT → 写入DB,分配server_id│  │
  │  │  ├── UPDATE → 检查版本:           │  │
  │  │  │   base_version == DB.version   │  │
  │  │  │   → 无冲突,更新成功,version+1 │  │
  │  │  │   base_version < DB.version    │  │
  │  │  │   → 冲突!返回服务器最新数据    │  │
  │  │  ├── 双式记账的两条记录原子写入   │  │
  │  │  └── 触发器自动写入change_log     │  │
  │  │                                    │  │
  │  │  响应:                             │  │
  │  │  {                                 │  │
  │  │    accepted: [                     │  │
  │  │      {                             │  │
  │  │        local_id,                   │  │
  │  │        entry_id,                   │  │
  │  │        debit_server_id,            │  │
  │  │        credit_server_id,           │  │
  │  │        seq                         │  │
  │  │      }                             │  │
  │  │    ],                              │  │
  │  │    conflicts: [                    │  │
  │  │      {                             │  │
  │  │        local_id,                   │  │
  │  │        server_record: {...},       │  │
  │  │        your_base_version: 3,       │  │
  │  │        server_version: 5           │  │
  │  │      }                             │  │
  │  │    ],                              │  │
  │  │    current_seq: 50                 │  │
  │  │  }                                 │  │
  │  └────────────────────────────────────┘  │
  │                                         │
  │  ┌──── 阶段2: PULL(下载服务器变更) ──┐  │
  │  │                                    │  │
  │  │  GET /api/sync/pull?since_seq=42   │  │
  │  │                                    │  │
  │  │  服务器查询:                       │  │
  │  │  SELECT * FROM sync_change_log     │  │
  │  │  WHERE family_id = ? AND seq > 42  │  │
  │  │  ORDER BY seq                      │  │
  │  │                                    │  │
  │  │  响应:                             │  │
  │  │  {                                 │  │
  │  │    changes: [                      │  │
  │  │      {seq:43, table, record_id,    │  │
  │  │       operation, version, data},   │  │
  │  │      {seq:44, ...},                │  │
  │  │      ...                           │  │
  │  │    ],                              │  │
  │  │    current_seq: 50,                │  │
  │  │    has_more: false                 │  │
  │  │  }                                 │  │
  │  │                                    │  │
  │  │  客户端按seq顺序回放:             │  │
  │  │  ├── INSERT → INSERT到本地SQLite  │  │
  │  │  ├── UPDATE → 用服务器数据覆盖    │  │
  │  │  └── DELETE → 软删除本地记录      │  │
  │  │  最后更新 last_pulled_seq = 50    │  │
  │  └────────────────────────────────────┘  │
  │                                         │
```

#### 16.5 冲突检测与解决

```
冲突场景:用户A(手机)和用户B(网页)同时编辑同一笔交易

手机端(base_version=3):  amount = 50
网页端(base_version=3):  amount = 80  ← 先提交成功,服务器version变为4

手机端push时:
  ├── 提交 {record_id:1001, base_version:3, amount:50}
  ├── 服务器检查:DB中version=4 ≠ base_version=3
  ├── 判定为冲突
  └── 返回: {conflict:true, server_record:{amount:80, version:4}}

冲突解决策略(分三级):
├── 策略1-自动(默认):以服务器为准,客户端覆盖本地
│   └── 适用:大部分场景,特别是另一端是"导入账单"等更完整的操作
│
├── 策略2-通知:以服务器为准,但通知用户有冲突
│   └── 适用:金额被覆盖时,提醒用户检查
│
└── 策略3-交互(未来):弹窗让用户选择保留哪个版本
    └── 适用:两端都是手动编辑的重要记录
```

#### 16.6 双式记账的原子同步

```
问题:一笔交易在DB中是2条记录(debit+credit),必须原子同步

解决方案:客户端按entry_id分组,服务器用事务批量写入

客户端push时:
  changes: [
    { table:"transactions", local_id:"a", entry_id:"e1", entry_side:"debit",  ... },
    { table:"transactions", local_id:"b", entry_id:"e1", entry_side:"credit", ... }
  ]

服务器处理:
  BEGIN TRANSACTION;
    INSERT INTO transactions (...debit...) RETURNING id → server_id_a;
    INSERT INTO transactions (...credit...) RETURNING id → server_id_b;
    -- 两条记录的entry_id指向同一个服务器端ID
    UPDATE transactions SET entry_id = server_id_a WHERE id = server_id_a;
    UPDATE transactions SET entry_id = server_id_a WHERE id = server_id_b;
  COMMIT;
  -- 触发器为两条记录各写一条change_log(seq不同但相邻)
```


```
├── 2. 网络恢复时 → 自动push+pull(监听connectivity变化)
```

```python
from apscheduler.schedulers.asyncio import AsyncIOScheduler
import subprocess, os, datetime

scheduler = AsyncIOScheduler()

# ==================== 数据库全量备份 ====================

@scheduler.scheduled_job('cron', hour=3, minute=0)  # 每天凌晨3点
async def backup_database():
    """PostgreSQL 全量备份"""
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    backup_dir = '/backup/db'
    os.makedirs(backup_dir, exist_ok=True)

    filename = f'{backup_dir}/family_finance_{timestamp}.sql.gz'

    # pg_dump + gzip 压缩
    cmd = f'pg_dump -h localhost -U postgres family_finance | gzip > {filename}'
    subprocess.run(cmd, shell=True, check=True)

    file_size = os.path.getsize(filename)

    # 记录备份日志
    await db.execute("""
        INSERT INTO backup_logs (backup_type, backup_target, file_path, file_size, file_format, status)
        VALUES ('db_full', 'local', $1, $2, 'sql.gz', 'success')
    """, filename, file_size)

    # 推送到配置的远程目标
    configs = await db.fetch("""
        SELECT * FROM backup_configs WHERE backup_type='db_full' AND is_enabled=TRUE
    """)

    for config in configs:
        if config['target'] == 'email':
            await send_backup_email(filename, config['target_config'])
        elif config['target'] == 'webdav':
            await upload_to_webdav(filename, config['target_config'])

    # 清理过期备份
    await cleanup_old_backups()


# ==================== Excel报表备份 ====================

@scheduler.scheduled_job('cron', hour=4, minute=0, day_of_week='mon')  # 每周一凌晨4点
async def backup_excel():
    """导出Excel报表备份"""
    timestamp = datetime.now().strftime('%Y%m%d')
    backup_dir = '/backup/excel'
    os.makedirs(backup_dir, exist_ok=True)

    # 导出全部账单为Excel
    transactions = await db.fetch("SELECT * FROM transactions WHERE is_deleted=FALSE")
    categories = await db.fetch('SELECT * FROM categories')
    accounts = await db.fetch('SELECT * FROM payment_accounts')

    # 生成Excel文件(用 openpyxl)
    filename = f'{backup_dir}/family_finance_{timestamp}.xlsx'
    await generate_excel_report(filename, transactions, categories, accounts)

    # 推送到邮箱
    await send_backup_email(filename, {'to': 'my_backup@qq.com'})


# ==================== 清理过期备份 ====================

async def cleanup_old_backups():
    """清理超过保留期的备份文件"""
    configs = await db.fetch('SELECT * FROM backup_configs WHERE is_enabled=TRUE')

    for config in configs:
        retention_days = config['retention_days']
        cutoff_date = datetime.now() - timedelta(days=retention_days)

        # 查找过期的备份记录
        expired = await db.fetch("""
            SELECT file_path FROM backup_logs
            WHERE backup_type = $1 AND created_at < $2 AND status = 'success'
            ORDER BY created_at DESC
            OFFSET $3
        """, config['backup_type'], cutoff_date, config['max_backups'])

        for record in expired:
            if os.path.exists(record['file_path']):
                os.remove(record['file_path'])


# ==================== 手机端备份上传 ====================

@app.post("/api/backup/upload")
async def upload_mobile_backup(file: UploadFile, user: User = Depends(get_current_user)):
    """手机端上传SQLite备份文件"""
    timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
    backup_dir = f'/backup/mobile/{user.family_id}'
    os.makedirs(backup_dir, exist_ok=True)

    filepath = f'{backup_dir}/{user.id}_{timestamp}.db'
    with open(filepath, 'wb') as f:
        content = await file.read()
        f.write(content)

    await db.execute("""
        INSERT INTO backup_logs (backup_type, backup_target, file_path, file_size, status)
        VALUES ('mobile_db', 'server', $1, $2, 'success')
    """, filepath, len(content))

    return {"success": True, "file_path": filepath}
```

```sql
-- 删除一笔交易时,必须同时删除借方和贷方两条记录
-- 建议用数据库触发器或应用层事务保证

-- 方案1:数据库触发器(推荐)
-- 注意:entry_id 为 NULL 时(快速录入未生成双式记录),不会触发联动,这是正确行为
CREATE OR REPLACE FUNCTION sync_delete_transaction()
RETURNS TRIGGER AS $$
BEGIN
    -- 当一条记录被标记为删除时,自动标记同一 entry_id 的另一条
    -- entry_id 为 NULL 时不触发(快速录入场景)
    IF NEW.is_deleted = TRUE AND OLD.is_deleted = FALSE AND NEW.entry_id IS NOT NULL THEN
        UPDATE transactions
        SET is_deleted = TRUE, updated_at = NOW()
        WHERE entry_id = NEW.entry_id
          AND id != NEW.id
          AND is_deleted = FALSE;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sync_delete
    AFTER UPDATE ON transactions
    FOR EACH ROW
    EXECUTE FUNCTION sync_delete_transaction();
```

```python
async def push_to_server(mobile_record: dict) -> dict:
    """手机记录 → 服务器双式记录"""
    entry_id = await generate_entry_id()

    # 生成借方记录
    debit_id = await db.execute("""
        INSERT INTO transactions (entry_id, entry_side, type, amount, category_id,
            payment_account_id, payment_channel_id, platform_id, ...)
        VALUES ($1, 'debit', $2, $3, $4, $5, $6, $7, ...)
    """, entry_id, mobile_record['type'], mobile_record['amount'],
        mobile_record['category_id'], mobile_record['payment_account_id'],
        mobile_record['payment_channel_id'], mobile_record['platform_id'])

    # 生成贷方记录
    credit_id = await db.execute("""
        INSERT INTO transactions (entry_id, entry_side, type, amount,
            payment_account_id, ...)
        VALUES ($1, 'credit', $2, $3, $4, ...)
    """, entry_id, mobile_record['type'], mobile_record['amount'],
        mobile_record['payment_account_id'])

    return {'entry_id': entry_id, 'debit_id': debit_id, 'credit_id': credit_id}


async def pull_to_mobile(server_records: list) -> dict:
    """服务器双式记录 → 手机单条记录"""
    # 按 entry_id 分组
    entries = group_by_entry_id(server_records)

    mobile_records = []
    for entry_id, records in entries.items():
        debit = next(r for r in records if r['entry_side'] == 'debit')
        credit = next(r for r in records if r['entry_side'] == 'credit')

        mobile_records.append({
            'server_id': debit['id'],
            'entry_id': entry_id,
            'type': debit['type'],
            'amount': debit['amount'],
            'category_id': debit['category_id'],
            'payment_account_id': credit['payment_account_id'],
            'payment_channel_id': debit['payment_channel_id'],
            'platform_id': debit['platform_id'],
            'merchant_name': debit['merchant_name'],
            'transaction_time': debit['transaction_time'],
        })

    return mobile_records
```

```
     SELECT transaction_id FROM transaction_tags tt
     JOIN tags t ON tt.tag_id = t.id WHERE t.name = '可报销'
   ) AND id NOT IN (SELECT original_txn_id FROM transaction_refunds);
3. 创建报销单,关联这些消费
4. 提交报销 → 审批 → 到账
5. 统计:"还有多少报销款没收回来?"
   SELECT SUM(total_amount - COALESCE(received_amount,0)) FROM reimbursements WHERE status != 'received';
```

### 29. 预算支持按周(补充)

```sql
-- budgets 表增加 weekly 支持
-- period 字段已有 monthly / yearly,增加 weekly

-- 周预算示例:
INSERT INTO budgets (family_id, category_id, amount, period, year, month)
VALUES (1, (SELECT id FROM categories WHERE name='餐饮'), 800, 'weekly', 2026, NULL);

-- 周预算查询(当前周):
SELECT
    b.amount as 周预算,
    COALESCE(SUM(t.amount), 0) as 本周已花,
    b.amount - COALESCE(SUM(t.amount), 0) as 剩余
FROM budgets b
LEFT JOIN transactions t ON t.category_id = b.category_id
    AND t.type = 'expense'
    AND t.is_deleted = FALSE
    AND t.transaction_time >= date_trunc('week', CURRENT_DATE)
    AND t.transaction_time < date_trunc('week', CURRENT_DATE) + interval '7 days'
WHERE b.period = 'weekly'
    AND b.year = EXTRACT(YEAR FROM CURRENT_DATE)
GROUP BY b.id, b.amount;
```
